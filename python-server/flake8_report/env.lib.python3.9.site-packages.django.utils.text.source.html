447 def smart_split(text):
448 r"""
449 Generator that splits a string by spaces, leaving quoted phrases together.
450 Supports both single and double quotes, and supports escaping quotes with
451 backslashes. In the output, strings will keep their initial and trailing

453 be further processed with unescape_string_literal()).

455 >>> list(smart_split(r'This is "a person\'s" test.'))
456 ['This', 'is', '"a person\\\'s"', 'test.']

460 ['A', '"\\"funky\\" style"', 'test.']
