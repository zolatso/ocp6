43 def find_index(fks, id_):
44 """
45 Look for the id on fks, fks is an array of arrays, each array has on [1]
46 the id of the class in a dia diagram. When not present returns None, else
47 it returns the position of the class with id on fks
48 """
49 for i, _ in fks.items():
50 if fks[i][1] == id_:
51 return i
52 return None
