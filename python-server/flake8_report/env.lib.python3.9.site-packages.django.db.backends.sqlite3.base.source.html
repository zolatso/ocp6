309 class SQLiteCursorWrapper(Database.Cursor):
310 """
311 Django uses the "format" and "pyformat" styles, but Python's sqlite3 module
312 supports neither of these styles.
313  
314 This wrapper performs the following conversions:
315  
316 - "format" style to "qmark" style
317 - "pyformat" style to "named" style
318  
319 In both cases, if you want to use a literal "%s", you'll need to use "%%s".
320 """
321  
322 def execute(self, query, params=None):
323 if params is None:
324 return super().execute(query)
325 # Extract names if params is a mapping, i.e. "pyformat" style is used.
326 param_names = list(params) if isinstance(params, Mapping) else None
327 query = self.convert_query(query, param_names=param_names)
328 return super().execute(query, params)

330 def executemany(self, query, param_list):
331 # Extract names if params is a mapping, i.e. "pyformat" style is used.
332 # Peek carefully as a generator can be passed instead of a list/tuple.
333 peekable, param_list = tee(iter(param_list))
334 if (params := next(peekable, None)) and isinstance(params, Mapping):
335 param_names = list(params)
336 else:
337 param_names = None
338 query = self.convert_query(query, param_names=param_names)
339 return super().executemany(query, param_list)

341 def convert_query(self, query, *, param_names=None):
342 if param_names is None:
343 # Convert from "format" style to "qmark" style.
344 return FORMAT_QMARK_REGEX.sub("?", query).replace("%%", "%")
345 else:
346 # Convert from "pyformat" style to "named" style.
347 return query % {name: f":{name}" for name in param_names}
