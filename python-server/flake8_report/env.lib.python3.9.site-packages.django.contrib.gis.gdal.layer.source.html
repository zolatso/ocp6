75 def _make_feature(self, feat_id):
76 """
77 Helper routine for __getitem__ that constructs a Feature from the given
78 Feature ID. If the OGR Layer does not support random-access reading,
79 then each feature of the layer will be incremented through until the
80 a Feature is found matching the given feature ID.
81 """
82 if self._random_read:
83 # If the Layer supports random reading, return.
84 try:
85 return Feature(capi.get_feature(self.ptr, feat_id), self)
86 except GDALException:
87 pass
88 else:
89 # Random access isn't supported, have to increment through
90 # each feature until the given feature ID is encountered.
91 for feat in self:
92 if feat.fid == feat_id:
93 return feat
94 # Should have returned a Feature, raise an IndexError.
95 raise IndexError("Invalid feature id: %s." % feat_id)
