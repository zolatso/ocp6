56 def urlencode(query, doseq=False):
57 """
58 A version of Python's urllib.parse.urlencode() function that can operate on
59 MultiValueDict and non-string values.
60 """
61 if isinstance(query, MultiValueDict):
62 query = query.lists()
63 elif hasattr(query, "items"):
64 query = query.items()
65 query_params = []
66 for key, value in query:
67 if value is None:
68 raise TypeError(
69 "Cannot encode None for key '%s' in a query string. Did you "
70 "mean to pass an empty string or omit the value?" % key
71 )
72 elif not doseq or isinstance(value, (str, bytes)):
73 query_val = value
74 else:
75 try:
76 itr = iter(value)
77 except TypeError:
78 query_val = value
79 else:
80 # Consume generators and iterators, when doseq=True, to
81 # work around https://bugs.python.org/issue31706.
82 query_val = []
83 for item in itr:
84 if item is None:
85 raise TypeError(
86 "Cannot encode None for key '%s' in a query "
87 "string. Did you mean to pass an empty string or "
88 "omit the value?" % key
89 )
90 elif not isinstance(item, bytes):
91 item = str(item)
92 query_val.append(item)
93 query_params.append((key, query_val))
94 return original_urlencode(query_params, doseq)

111 def parse_http_date(date):
112 """
113 Parse a date format as specified by HTTP RFC 9110 Section 5.6.7.
114  
115 The three formats allowed by the RFC are accepted, even if only the first
116 one is still in widespread use.
117  
118 Return an integer expressed in seconds since the epoch, in UTC.
119 """
120 # email.utils.parsedate() does the job for RFC 1123 dates; unfortunately
121 # RFC 9110 makes it mandatory to support RFC 850 dates too. So we roll
122 # our own RFC-compliant parsing.
123 for regex in RFC1123_DATE, RFC850_DATE, ASCTIME_DATE:
124 m = regex.match(date)
125 if m is not None:
126 break
127 else:
128 raise ValueError("%r is not in a valid HTTP date format" % date)
129 try:
130 tz = datetime.timezone.utc
131 year = int(m["year"])
132 if year < 100:
133 current_year = datetime.datetime.now(tz=tz).year
134 current_century = current_year - (current_year % 100)
135 if year - (current_year % 100) > 50:
136 # year that appears to be more than 50 years in the future are
137 # interpreted as representing the past.
138 year += current_century - 100
139 else:
140 year += current_century
141 month = MONTHS.index(m["mon"].lower()) + 1
142 day = int(m["day"])
143 hour = int(m["hour"])
144 min = int(m["min"])
145 sec = int(m["sec"])
146 result = datetime.datetime(year, month, day, hour, min, sec, tzinfo=tz)
147 return int(result.timestamp())
148 except Exception as exc:
149 raise ValueError("%r is not a valid date" % date) from exc
