32 def slugify(value, separator, unicode=False):
33 """ Slugify a string, to make it URL friendly. """

36 value = unicodedata.normalize('NFKD', value)
37 value = value.encode('ascii', 'ignore').decode('ascii')
38 value = re.sub(r'[^\w\s-]', '', value).strip().lower()
39 return re.sub(r'[{}\s]+'.format(separator), separator, value)

97 def nest_toc_tokens(toc_list):
98 """Given an unsorted list with errors and skips, return a nested one.
99  
100 [{'level': 1}, {'level': 2}]
101 =>
102 [{'level': 1, 'children': [{'level': 2, 'children': []}]}]
103  
104 A wrong list is also converted:
105  
106 [{'level': 2}, {'level': 1}]
107 =>
108 [{'level': 2, 'children': []}, {'level': 1, 'children': []}]
109 """
110  
111 ordered_list = []
112 if len(toc_list):
113 # Initialize everything by processing the first entry
114 last = toc_list.pop(0)
115 last['children'] = []
116 levels = [last['level']]
117 ordered_list.append(last)
118 parents = []
119  
120 # Walk the rest nesting the entries properly
121 while toc_list:
122 t = toc_list.pop(0)
123 current_level = t['level']
124 t['children'] = []
125  
126 # Reduce depth if current level < last item's level
127 if current_level < levels[-1]:
128 # Pop last level since we know we are less than it
129 levels.pop()
130  
131 # Pop parents and levels we are less than or equal to
132 to_pop = 0
133 for p in reversed(parents):
134 if current_level <= p['level']:
135 to_pop += 1
136 else: # pragma: no cover
137 break
138 if to_pop:
139 levels = levels[:-to_pop]
140 parents = parents[:-to_pop]
141  
142 # Note current level as last
143 levels.append(current_level)
144  
145 # Level is the same, so append to
146 # the current parent (if available)
147 if current_level == levels[-1]:
148 (parents[-1]['children'] if parents
149 else ordered_list).append(t)
150  
151 # Current level is > last item's level,
152 # So make last item a parent and append current as child
153 else:
154 last['children'].append(t)
155 parents.append(last)
156 levels.append(current_level)
157 last = t
158  
159 return ordered_list
