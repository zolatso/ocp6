50 class SimpleRateThrottle(BaseThrottle):
51 """
52 A simple cache implementation, that only requires `.get_cache_key()`
53 to be overridden.
54  
55 The rate (requests / seconds) is set by a `rate` attribute on the Throttle

97 def parse_rate(self, rate):
98 """
99 Given the request rate string, return a two tuple of:
100 <allowed number of requests>, <period of time in seconds>
101 """
102 if rate is None:
103 return (None, None)
104 num, period = rate.split('/')
105 num_requests = int(num)
106 duration = {'s': 1, 'm': 60, 'h': 3600, 'd': 86400}[period[0]]
107 return (num_requests, duration)

109 def allow_request(self, request, view):
110 """
111 Implement the check to see if the request should be throttled.
112  
113 On success calls `throttle_success`.
114 On failure calls `throttle_failure`.
115 """
116 if self.rate is None:
117 return True
118  
119 self.key = self.get_cache_key(request, view)
120 if self.key is None:
121 return True
122  
123 self.history = self.cache.get(self.key, [])
124 self.now = self.timer()
125  
126 # Drop any requests from the history which have now passed the
127 # throttle duration
128 while self.history and self.history[-1] <= self.now - self.duration:
129 self.history.pop()
130 if len(self.history) >= self.num_requests:
131 return self.throttle_failure()
132 return self.throttle_success()
133  
134 def throttle_success(self):
135 """
136 Inserts the current request's timestamp along with the key
137 into the cache.
138 """
139 self.history.insert(0, self.now)
140 self.cache.set(self.key, self.history, self.duration)
141 return True

144 """
145 Called when a request to the API has failed due to throttling.

147 return False

149 def wait(self):
150 """
151 Returns the recommended next request time in seconds.
152 """
153 if self.history:
154 remaining_duration = self.duration - (self.now - self.history[-1])
155 else:
156 remaining_duration = self.duration
157  
158 available_requests = self.num_requests - len(self.history) + 1
159 if available_requests <= 0:
160 return None
161  
162 return remaining_duration / float(available_requests)
