138 def convert_extent(self, clob):
139 if clob:
140 # Generally, Oracle returns a polygon for the extent -- however,
141 # it can return a single point if there's only one Point in the
142 # table.
143 ext_geom = GEOSGeometry(memoryview(clob.read()))
144 gtype = str(ext_geom.geom_type)
145 if gtype == "Polygon":
146 # Construct the 4-tuple from the coordinates in the polygon.
147 shell = ext_geom.shell
148 ll, ur = shell[0][:2], shell[2][:2]
149 elif gtype == "Point":
150 ll = ext_geom.coords[:2]
151 ur = ll
152 else:
153 raise Exception(
154 "Unexpected geometry type returned for extent: %s" % gtype
155 )
156 xmin, ymin = ll
157 xmax, ymax = ur
158 return (xmin, ymin, xmax, ymax)
159 else:
160 return None

170 def get_distance(self, f, value, lookup_type):
171 """
172 Return the distance parameters given the value and the lookup type.
173 On Oracle, geometry columns with a geodetic coordinate system behave
174 implicitly like a geography column, and thus meters will be used as
175 the distance parameter on them.
176 """
177 if not value:
178 return []
179 value = value[0]

181 if f.geodetic(self.connection):
182 dist_param = value.m
183 else:
184 dist_param = getattr(
185 value, Distance.unit_attname(f.units_name(self.connection))
186 )
187 else:
188 dist_param = value

191 # that starts with "distance=".
192 if lookup_type == "dwithin":
193 dist_param = "distance=%s" % dist_param
194  
195 return [dist_param]
