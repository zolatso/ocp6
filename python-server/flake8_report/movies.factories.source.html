19 def _get_related(related_model, related_factory, number=3, total=None):
20 """Handles extracted arguments for related instances."""
21 count = related_model.objects.count()
22 total = 2 * number if total is None else max(number, total)
23 if count < total:
24 for _ in range(total - count):
25 related_factory()
26 return related_model.objects.order_by("?")[:number]
