134 def check(source_dir):
135 pyproject = pjoin(source_dir, "pyproject.toml")
136 if isfile(pyproject):
137 log.info("Found pyproject.toml")
138 else:
139 log.error("Missing pyproject.toml")
140 return False
141  
142 try:
143 with open(pyproject) as f:
144 pyproject_data = toml_load(f)
145 # Ensure the mandatory data can be loaded
146 buildsys = pyproject_data["build-system"]
147 requires = buildsys["requires"]
148 backend = buildsys["build-backend"]
149 backend_path = buildsys.get("backend-path")
150 log.info("Loaded pyproject.toml")
151 except (TomlDecodeError, KeyError):
152 log.error("Invalid pyproject.toml", exc_info=True)
153 return False
154  
155 hooks = Pep517HookCaller(source_dir, backend, backend_path)
156  
157 sdist_ok = check_build_sdist(hooks, requires)
158 wheel_ok = check_build_wheel(hooks, requires)
159  
160 if not sdist_ok:
161 log.warning("Sdist checks failed; scroll up to see")
162 if not wheel_ok:
163 log.warning("Wheel checks failed")
164  
165 return sdist_ok
