52 def evaluate(self, expr, context):
53 """
54 Evaluate a marker expression returned by the :func:`parse_requirement`
55 function in the specified context.
56 """
57 if isinstance(expr, string_types):
58 if expr[0] in "'\"":
59 result = expr[1:-1]
60 else:
61 if expr not in context:
62 raise SyntaxError("unknown variable: %s" % expr)
63 result = context[expr]
64 else:
65 assert isinstance(expr, dict)
66 op = expr["op"]
67 if op not in self.operations:
68 raise NotImplementedError("op not implemented: %s" % op)
69 elhs = expr["lhs"]
70 erhs = expr["rhs"]
71 if _is_literal(expr["lhs"]) and _is_literal(expr["rhs"]):

74 lhs = self.evaluate(elhs, context)
75 rhs = self.evaluate(erhs, context)
76 result = self.operations[op](lhs, rhs)
