190 def _parseHeader(self):
191 """
192 Determines language of a code block from shebang line and whether the
193 said line should be removed or left in place. If the shebang line
194 contains a path (even a single /) then it is assumed to be a real
195 shebang line and left alone. However, if no path is given
  • E501 Line too long (81 > 79 characters)
196 (e.i.: `#!python` or `:::python`) then it is assumed to be a mock shebang
197 for language identification of a code fragment and removed from the
198 code block prior to processing for code highlighting. When a mock
199 shebang (e.i: `#!python`) is found, line numbering is turned on. When
200 colons are found in place of a shebang (e.i.: `:::python`), line
201 numbering is left in the current state - off by default.
202  
203 Also parses optional list of highlight lines, like:
204  
205 :::python hl_lines="1 3"
206 """
207  
208 import re
209  
210 # split text into lines
211 lines = self.src.split("\n")
212 # pull first line to examine
213 fl = lines.pop(0)
214  
215 c = re.compile(r'''
216 (?:(?:^::+)|(?P<shebang>^[#]!)) # Shebang or 2 or more colons
217 (?P<path>(?:/\w+)*[/ ])? # Zero or 1 path
218 (?P<lang>[\w#.+-]*) # The language
219 \s* # Arbitrary whitespace
220 # Optional highlight lines, single- or double-quote-delimited
221 (hl_lines=(?P<quot>"|')(?P<hl_lines>.*?)(?P=quot))?
222 ''', re.VERBOSE)
223 # search first line for shebang
224 m = c.search(fl)
225 if m:
226 # we have a match
227 try:
228 self.lang = m.group('lang').lower()
229 except IndexError: # pragma: no cover
230 self.lang = None
231 if m.group('path'):
232 # path exists - restore first line
233 lines.insert(0, fl)
234 if self.options['linenos'] is None and m.group('shebang'):
235 # Overridable and Shebang exists - use line numbers
236 self.options['linenos'] = True
237  
238 self.options['hl_lines'] = parse_hl_lines(m.group('hl_lines'))
239 else:
240 # No match
241 lines.insert(0, fl)
242  
243 self.src = "\n".join(lines).strip("\n")

252 def code_unescape(self, text):
253 """Unescape code."""
254 text = text.replace("&lt;", "<")
255 text = text.replace("&gt;", ">")
256 # Escaped '&' should be replaced at the end to avoid
257 # conflicting with < and >.
258 text = text.replace("&amp;", "&")
259 return text

261 def run(self, root):
262 """ Find code blocks and store in `htmlStash`. """
263 blocks = root.iter('pre')
264 for block in blocks:
265 if len(block) == 1 and block[0].tag == 'code':
266 local_config = self.config.copy()
267 code = CodeHilite(
268 self.code_unescape(block[0].text),
269 tab_length=self.md.tab_length,
270 style=local_config.pop('pygments_style', 'default'),
271 **local_config
272 )
273 placeholder = self.md.htmlStash.store(code.hilite())
274 # Clear code block in `etree` instance
275 block.clear()
276 # Change to `p` element which will later
277 # be removed when inserting raw html
278 block.tag = 'p'
279 block.text = placeholder
