71 def get_available_name(self, name, max_length=None):
72 """
73 Return a filename that's free on the target storage system and
74 available for new content to be written to.
75 """
76 name = str(name).replace("\\", "/")
77 dir_name, file_name = os.path.split(name)
78 if ".." in pathlib.PurePath(dir_name).parts:
79 raise SuspiciousFileOperation(
80 "Detected path traversal attempt in '%s'" % dir_name
81 )
82 validate_file_name(file_name)
83 file_root, file_ext = os.path.splitext(file_name)
84 # If the filename already exists, generate an alternative filename
85 # until it doesn't exist.
86 # Truncate original name if required, so the new filename does not
87 # exceed the max_length.
88 while self.exists(name) or (max_length and len(name) > max_length):
89 # file_ext includes the dot.
90 name = os.path.join(
91 dir_name, self.get_alternative_name(file_root, file_ext)
92 )
93 if max_length is None:
94 continue
95 # Truncate file_root if max_length exceeded.
96 truncation = len(name) - max_length
97 if truncation > 0:
98 file_root = file_root[:-truncation]
99 # Entire file_root was truncated in attempt to find an
100 # available filename.
101 if not file_root:
102 raise SuspiciousFileOperation(
103 'Storage can not find an available filename for "%s". '
104 "Please make sure that the corresponding file field "
105 'allows sufficient "max_length".' % name
106 )
107 name = os.path.join(
108 dir_name, self.get_alternative_name(file_root, file_ext)
109 )
110 return name
