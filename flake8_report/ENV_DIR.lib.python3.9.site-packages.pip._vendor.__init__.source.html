27 # Define a small helper function to alias our vendored modules to the real ones
28 # if the vendored ones do not exist. This idea of this was taken from
29 # https://github.com/kennethreitz/requests/pull/2567.
30 def vendored(modulename):
31 vendored_name = "{0}.{1}".format(__name__, modulename)
32  
33 try:
34 __import__(modulename, globals(), locals(), level=0)
35 except ImportError:
36 # We can just silently allow import failures to pass here. If we
37 # got to this point it means that ``import pip._vendor.whatever``
38 # failed and so did ``import whatever``. Since we're importing this
39 # upfront in an attempt to alias imports, not erroring here will
40 # just mean we get a regular import error whenever pip *actually*
41 # tries to import one of these modules to use it, which actually
42 # gives us a better error message than we would have otherwise
43 # gotten.
44 pass
45 else:
46 sys.modules[vendored_name] = sys.modules[modulename]
47 base, head = vendored_name.rsplit(".", 1)
48 setattr(sys.modules[base], head, sys.modules[modulename])

54 # however downstream redistributors can enable it in a consistent way across
