376 @cached_property
377 def supports_transactions(self):
378 """Confirm support for transactions."""
379 with self.connection.cursor() as cursor:
380 cursor.execute("CREATE TABLE ROLLBACK_TEST (X INT)")
381 self.connection.set_autocommit(False)
382 cursor.execute("INSERT INTO ROLLBACK_TEST (X) VALUES (8)")
383 self.connection.rollback()
384 self.connection.set_autocommit(True)
385 cursor.execute("SELECT COUNT(X) FROM ROLLBACK_TEST")
386 (count,) = cursor.fetchone()
387 cursor.execute("DROP TABLE ROLLBACK_TEST")
388 return count == 0
