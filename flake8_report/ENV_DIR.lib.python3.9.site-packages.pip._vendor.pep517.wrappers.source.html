82 def norm_and_check(source_tree, requested):
83 """Normalise and check a backend path.
84  
85 Ensure that the requested backend path is specified as a relative path,
86 and resolves to a location under the given source tree.
87  
88 Return an absolute version of the requested path.
89 """
90 if os.path.isabs(requested):
91 raise ValueError("paths must be relative")
92  
93 abs_source = os.path.abspath(source_tree)
94 abs_requested = os.path.normpath(os.path.join(abs_source, requested))
95 # We have to use commonprefix for Python 2.7 compatibility. So we
96 # normalise case to avoid problems because commonprefix is a character
97 # based comparison :-(
98 norm_source = os.path.normcase(abs_source)
99 norm_requested = os.path.normcase(abs_requested)
100 if os.path.commonprefix([norm_source, norm_requested]) != norm_source:
101 raise ValueError("paths must be inside source tree")
102  
103 return abs_requested

249 def _call_hook(self, hook_name, kwargs):
250 # On Python 2, pytoml returns Unicode values (which is correct) but the
251 # environment passed to check_call needs to contain string values. We
252 # convert here by encoding using ASCII (the backend can only contain
253 # letters, digits and _, . and : characters, and will be used as a
254 # Python identifier, so non-ASCII content is wrong on Python 2 in
255 # any case).
256 # For backend_path, we use sys.getfilesystemencoding.
257 if sys.version_info[0] == 2:
258 build_backend = self.build_backend.encode("ASCII")
259 else:

261 extra_environ = {"PEP517_BUILD_BACKEND": build_backend}

264 backend_path = os.pathsep.join(self.backend_path)
265 if sys.version_info[0] == 2:
266 backend_path = backend_path.encode(sys.getfilesystemencoding())
