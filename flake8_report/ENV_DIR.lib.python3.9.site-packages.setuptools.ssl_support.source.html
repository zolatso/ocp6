63 def _dnsname_match(dn, hostname, max_wildcards=1):
64 """Matching according to RFC 6125, section 6.4.3
65  
66 https://tools.ietf.org/html/rfc6125#section-6.4.3
67 """
68 pats = []
69 if not dn:
70 return False
71  
72 # Ported from python3-syntax:
73 # leftmost, *remainder = dn.split(r'.')
74 parts = dn.split(r".")
75 leftmost = parts[0]
76 remainder = parts[1:]
77  
78 wildcards = leftmost.count("*")
79 if wildcards > max_wildcards:
80 # Issue #17980: avoid denials of service by refusing more
81 # than one wildcard per fragment. A survey of established
82 # policy among SSL implementations showed it to be a
83 # reasonable choice.
84 raise CertificateError(
85 "too many wildcards in certificate DNS name: " + repr(dn)
86 )
87  
88 # speed up common case w/o wildcards
89 if not wildcards:
90 return dn.lower() == hostname.lower()
91  
92 # RFC 6125, section 6.4.3, subitem 1.
93 # The client SHOULD NOT attempt to match a
94 # presented identifier in which the wildcard
95 # character comprises a label other than the
96 # left-most label.
97 if leftmost == "*":
98 # When '*' is a fragment by itself, it matches a non-empty dotless
99 # fragment.
100 pats.append("[^.]+")
101 elif leftmost.startswith("xn--") or hostname.startswith("xn--"):
102 # RFC 6125, section 6.4.3, subitem 3.
103 # The client SHOULD NOT attempt to match a presented identifier
104 # where the wildcard character is embedded within an A-label or
105 # U-label of an internationalized domain name.
106 pats.append(re.escape(leftmost))
107 else:
108 # Otherwise, '*' matches any dotless string, e.g. www*
109 pats.append(re.escape(leftmost).replace(r"\*", "[^.]*"))
110  
111 # add the remaining fragments, ignore any wildcards
112 for frag in remainder:
113 pats.append(re.escape(frag))
114  
115 pat = re.compile(r"\A" + r"\.".join(pats) + r"\Z", re.IGNORECASE)
116 return pat.match(hostname)
