35  
36 `progress_filter` is a function taking two arguments: a source path

38 will be extracted. The callback must return the desired extract path
39 (which may be the same as the one passed in), or else ``None`` to skip
40 that file or directory. The callback can thus be used to report on the
41 progress of the extraction, as well as to filter the items extracted or

52 order.

65 def unpack_directory(filename, extract_dir, progress_filter=default_filter):
66 """ "Unpack" a directory, using the same interface as for archives
67  
68 Raises ``UnrecognizedFormat`` if `filename` is not a directory
69 """
70 if not os.path.isdir(filename):
71 raise UnrecognizedFormat("%s is not a directory" % filename)
72  
73 paths = {
74 filename: ("", extract_dir),
75 }
76 for base, dirs, files in os.walk(filename):
77 src, dst = paths[base]
78 for d in dirs:
79 paths[os.path.join(base, d)] = src + d + "/", os.path.join(dst, d)
80 for f in files:
81 target = os.path.join(dst, f)
82 target = progress_filter(src + f, target)
83 if not target:
84 # skip non-files
85 continue
86 ensure_directory(target)
87 f = os.path.join(base, f)
88 shutil.copyfile(f, target)
89 shutil.copystat(f, target)

150 def _iter_open_tar(tar_obj, extract_dir, progress_filter):
151 """Emit member-destination pairs from a tar archive."""
152 # don't do any chowning!
153 tar_obj.chown = lambda *args: None
154  
155 with contextlib.closing(tar_obj):
156 for member in tar_obj:
157 name = member.name
158 # don't extract absolute paths or ones with .. in them
159 if name.startswith("/") or ".." in name.split("/"):
160 continue
161  
162 prelim_dst = os.path.join(extract_dir, *name.split("/"))
163  
164 try:
165 member = _resolve_tar_file_or_dir(tar_obj, member)
166 except LookupError:
167 continue

170 if not final_dst:
171 continue

174 final_dst = final_dst[:-1]
