36 def check_string(result, func, cargs, offset=-1, str_result=False):
37 """
38 Check the string output returned from the given function, and free
39 the string pointer allocated by OGR. The `str_result` keyword
40 may be used when the result is the string pointer, otherwise
41 the OGR error code is assumed. The `offset` keyword may be used
42 to extract the string pointer passed in by-reference at the given
43 slice offset in the function arguments.
44 """
45 if str_result:
46 # For routines that return a string.
47 ptr = result
48 if not ptr:
49 s = None
50 else:
51 s = string_at(result)
52 else:
53 # Error-code return specified.
54 check_err(result)
55 ptr = ptr_byref(cargs, offset)
56 # Getting the string value
57 s = ptr.value
58 # Correctly freeing the allocated memory behind GDAL pointer
59 # with the VSIFree routine.
60 if ptr:
61 lgdal.VSIFree(ptr)
62 return s
