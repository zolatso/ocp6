51 def _handle_word(s, t):
52 if t.startswith('.'):
53 return '.', t[1:]
54 if t.startswith('#'):
55 return 'id', t[1:]
56 return t, t

149 def assign_attrs(self, elem: Element, attrs: str) -> None:
150 """ Assign `attrs` to element. """
151 for k, v in get_attrs(attrs):
152 if k == '.':
153 # add to class
154 cls = elem.get('class')
155 if cls:
156 elem.set('class', '{} {}'.format(cls, v))
157 else:
158 elem.set('class', v)
159 else:
160 # assign attribute `k` with `v`
161 elem.set(self.sanitize_name(k), v)

163 def sanitize_name(self, name: str) -> str:
164 """
165 Sanitize name as 'an XML Name, minus the ":"'.
166 See https://www.w3.org/TR/REC-xml-names/#NT-NCName
167 """
168 return self.NAME_RE.sub('_', name)
