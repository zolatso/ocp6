951 @contextmanager
952 def timed(self, name):
953 self.records[name]

955 try:
956 yield

958 end_time = time.perf_counter() - start_time

962 for name, end_times in self.records.items():
963 for record_time in end_times:
964 record = "%s took %.3fs" % (name, record_time)
