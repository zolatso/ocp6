53 class CodeNode(Node):
54 def __init__(self, language, nodelist, name=''):
55 self.language = Variable(language)
56 self.nodelist = nodelist
57 if name:
58 self.name = Variable(name)
59 else:
60 self.name = None
61  
62 def render(self, context):
63 code = self.nodelist.render(context).strip()
64 lexer = get_lexer_by_name(self.language.resolve(context))
65 formatter = HtmlFormatter(linenos=False)
66 html = ""
67 if self.name:
68 name = self.name.resolve(context)
69 html = '<div class="predesc"><span>%s</span></div>' % name
70 return html + pyghighlight(code, lexer, formatter)
