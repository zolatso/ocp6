147 def create_command(name, **kwargs):
148 # type: (str, **Any) -> Command
149 """
150 Create an instance of the Command class with the given name.
151 """
152 module_path, class_name, summary = commands_dict[name]
153 module = importlib.import_module(module_path)
154 command_class = getattr(module, class_name)
155 command = command_class(name=name, summary=summary, **kwargs)
156  
157 return command
