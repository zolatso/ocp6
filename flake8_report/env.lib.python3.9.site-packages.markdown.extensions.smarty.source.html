95 # Constants for quote education.
96 punctClass = r"""[!"#\$\%'()*+,-.\/:;<=>?\@\[\\\]\^_`{|}~]"""
97 endOfWordClass = r"[\s.,;:!?)]"
98 closeClass = r"[^\ \t\r\n\[\{\(\-\u0002\u0003]"
99  
100 openingQuotesBase = (
101 r'(\s' # a whitespace char
102 r'|&nbsp;' # or a non-breaking space entity
103 r'|--' # or dashes
104 r'|–|—' # or Unicode
105 r'|&[mn]dash;' # or named dash entities
106 r'|&#8211;|&#8212;' # or decimal entities
107 r')'
108 )

116 'left-single-quote': '&lsquo;',
117 'right-single-quote': '&rsquo;',
118 'left-double-quote': '&ldquo;',

123 # Special case if the very first character is a quote
124 # followed by punctuation at a non-word-break. Close the quotes by brute force:
125 singleQuoteStartRe = r"^'(?=%s\B)" % punctClass
126 doubleQuoteStartRe = r'^"(?=%s\B)' % punctClass
127  
128 # Special case for double sets of quotes, e.g.:
129 # <p>He said, "'Quoted' words in a larger quote."</p>
130 doubleQuoteSetsRe = r""""'(?=\w)"""
131 singleQuoteSetsRe = r"""'"(?=\w)"""
132  
133 # Special case for decade abbreviations (the '80s):
134 decadeAbbrRe = r"(?<!\w)'(?=\d{2}s)"
135  
136 # Get most opening double quotes:
137 openingDoubleQuotesRegex = r'%s"(?=\w)' % openingQuotesBase
138  
139 # Double closing quotes:
140 closingDoubleQuotesRegex = r'"(?=\s)'
141 closingDoubleQuotesRegex2 = '(?<=%s)"' % closeClass
142  
143 # Get most opening single quotes:
144 openingSingleQuotesRegex = r"%s'(?=\w)" % openingQuotesBase
145  
146 # Single closing quotes:
147 closingSingleQuotesRegex = r"(?<=%s)'(?!\s|s\b|\d)" % closeClass
148 closingSingleQuotesRegex2 = r"'(\s|s\b)"
149  
150 # All remaining quotes should be opening ones
151 remainingSingleQuotesRegex = r"'"
152 remainingDoubleQuotesRegex = r'"'

154 HTML_STRICT_RE = HTML_RE + r'(?!\>)'

157 class SubstituteTextPattern(HtmlInlineProcessor):
158 def __init__(self, pattern, replace, md):
159 """ Replaces matches with some text. """
160 HtmlInlineProcessor.__init__(self, pattern)
161 self.replace = replace
162 self.md = md
163  
164 def handleMatch(self, m, data):
165 result = ''
166 for part in self.replace:
167 if isinstance(part, int):
168 result += m.group(part)
169 else:
170 result += self.md.htmlStash.store(part)
171 return result, m.start(0), m.end(0)

174 class SmartyExtension(Extension):
175 """ Add Smarty to Markdown. """
176 def __init__(self, **kwargs):
177 self.config = {
178 'smart_quotes': [True, 'Educate quotes'],
179 'smart_angled_quotes': [False, 'Educate angled quotes'],
180 'smart_dashes': [True, 'Educate dashes'],
181 'smart_ellipses': [True, 'Educate ellipses'],
182 'substitutions': [{}, 'Overwrite default substitutions'],
183 }
184 """ Default configuration options. """
185 super().__init__(**kwargs)
186 self.substitutions = dict(substitutions)
187 self.substitutions.update(self.getConfig('substitutions', default={}))

189 def _addPatterns(self, md, patterns, serie, priority):
190 for ind, pattern in enumerate(patterns):
191 pattern += (md,)
192 pattern = SubstituteTextPattern(*pattern)
193 name = 'smarty-%s-%d' % (serie, ind)
194 self.inlinePatterns.register(pattern, name, priority-ind)

196 def educateDashes(self, md) -> None:
197 emDashesPattern = SubstituteTextPattern(
198 r'(?<!-)---(?!-)', (self.substitutions['mdash'],), md
199 )
200 enDashesPattern = SubstituteTextPattern(
201 r'(?<!-)--(?!-)', (self.substitutions['ndash'],), md
202 )
203 self.inlinePatterns.register(emDashesPattern, 'smarty-em-dashes', 50)
204 self.inlinePatterns.register(enDashesPattern, 'smarty-en-dashes', 45)

206 def educateEllipses(self, md) -> None:
207 ellipsesPattern = SubstituteTextPattern(

209 )
210 self.inlinePatterns.register(ellipsesPattern, 'smarty-ellipses', 10)

222 def educateQuotes(self, md) -> None:
223 lsquo = self.substitutions['left-single-quote']
224 rsquo = self.substitutions['right-single-quote']
225 ldquo = self.substitutions['left-double-quote']

244 def extendMarkdown(self, md):
245 configs = self.getConfigs()
246 self.inlinePatterns: Registry[HtmlInlineProcessor] = Registry()
247 if configs['smart_ellipses']:
248 self.educateEllipses(md)
249 if configs['smart_quotes']:
250 self.educateQuotes(md)
251 if configs['smart_angled_quotes']:
252 self.educateAngledQuotes(md)

257 self.educateDashes(md)
