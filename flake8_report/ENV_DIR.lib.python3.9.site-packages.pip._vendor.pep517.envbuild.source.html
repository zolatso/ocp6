131 def build_wheel(source_dir, wheel_dir, config_settings=None):
132 """Build a wheel from a source directory using PEP 517 hooks.
133  
134 :param str source_dir: Source directory containing pyproject.toml
135 :param str wheel_dir: Target directory to create wheel in
136 :param dict config_settings: Options to pass to build backend
137  
138 This is a blocking function which will run pip in a subprocess to install
139 build requirements.
140 """
141 if config_settings is None:
142 config_settings = {}
143 requires, backend, backend_path = _load_pyproject(source_dir)
144 hooks = Pep517HookCaller(source_dir, backend, backend_path)
145  
146 with BuildEnvironment() as env:
147 env.pip_install(requires)
148 reqs = hooks.get_requires_for_build_wheel(config_settings)
149 env.pip_install(reqs)
150 return hooks.build_wheel(wheel_dir, config_settings)
