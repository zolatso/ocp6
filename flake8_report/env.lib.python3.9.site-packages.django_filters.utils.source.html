32 """
33 Handles the deprecation paths when renaming an attribute.
34  
35 It does the following:
36 - Defines accessors that redirect to the renamed attributes.
37 - Complain whenever an old attribute is accessed.

39 This is conceptually based on `django.utils.deprecation.RenameMethodsBase`.

44 def __new__(metacls, name, bases, attrs):
45 # remove old attributes before creating class
46 old_names = [r[0] for r in metacls.renamed_attributes]
47 old_names = [name for name in old_names if name in attrs]
48 old_attrs = {name: attrs.pop(name) for name in old_names}
49  
50 # get a handle to any accessors defined on the class
51 cls_getattr = attrs.pop("__getattr__", None)
52 cls_setattr = attrs.pop("__setattr__", None)
53  
54 new_class = super().__new__(metacls, name, bases, attrs)

56 def __getattr__(self, name):
57 name = type(self).get_name(name)
58 if cls_getattr is not None:
59 return cls_getattr(self, name)
60 elif hasattr(super(new_class, self), "__getattr__"):
61 return super(new_class, self).__getattr__(name)
62 return self.__getattribute__(name)
63  
64 def __setattr__(self, name, value):
65 name = type(self).get_name(name)
66 if cls_setattr is not None:
67 return cls_setattr(self, name, value)
68 return super(new_class, self).__setattr__(name, value)
69  
70 new_class.__getattr__ = __getattr__
71 new_class.__setattr__ = __setattr__
72  
73 # set renamed attributes
74 for name, value in old_attrs.items():
75 setattr(new_class, name, value)

150 def get_field_parts(model, field_name):
151 """
152 Get the field parts that represent the traversable relationships from the
153 base ``model`` to the final field, described by ``field_name``.
154  
155 ex::
156  
157 >>> parts = get_field_parts(Book, 'author__first_name')
158 >>> [p.verbose_name for p in parts]
159 ['author', 'first name']
160  
161 """
162 parts = field_name.split(LOOKUP_SEP)
163 opts = model._meta
164 fields = []
165  
166 # walk relationships
167 for name in parts:
168 try:
169 field = opts.get_field(name)
170 except FieldDoesNotExist:
171 return None
172  
173 fields.append(field)
174 try:
175 if isinstance(field, RelatedField):
176 opts = field.remote_field.model._meta
177 elif isinstance(field, ForeignObjectRel):
178 opts = field.related_model._meta
179 except AttributeError:
180 # Lazy relationships are not resolved until registry is populated.
181 raise RuntimeError(
182 "Unable to resolve relationship `%s` for `%s`. Django is most "
183 "likely not initialized, and its apps registry not populated. "
184 "Ensure Django has finished setup before loading `FilterSet`s."
185 % (field_name, model._meta.label)
186 )
187  
188 return fields

191 def resolve_field(model_field, lookup_expr):
192 """
193 Resolves a ``lookup_expr`` into its final output field, given
194 the initial ``model_field``. The lookup expression should only contain
195 transforms and lookups, not intermediary model field parts.
196  
197 Note:
198 This method is based on django.db.models.sql.query.Query.build_lookup
199  
200 For more info on the lookup API:
201 https://docs.djangoproject.com/en/stable/ref/models/lookups/
202  
203 """
204 query = model_field.model._default_manager.all().query
205 lhs = Expression(model_field)
206 lookups = lookup_expr.split(LOOKUP_SEP)
207  
208 assert len(lookups) > 0
209  
210 try:
211 while lookups:
212 name = lookups[0]
213 args = (lhs, name)
214 # If there is just one part left, try first get_lookup() so
215 # that if the lhs supports both transform and lookup for the
216 # name, then lookup will be picked.
217 if len(lookups) == 1:
218 final_lookup = lhs.get_lookup(name)
219 if not final_lookup:
220 # We didn't find a lookup. We are going to interpret
221 # the name as transform, and do an Exact lookup against
222 # it.
223 lhs = query.try_transform(*args)
224 final_lookup = lhs.get_lookup("exact")
225 return lhs.output_field, final_lookup.lookup_name
226 lhs = query.try_transform(*args)
227 lookups = lookups[1:]
228 except FieldError as e:
229 raise FieldLookupError(model_field, lookup_expr) from e

253 def verbose_field_name(model, field_name):
254 """
255 Get the verbose name for a given ``field_name``. The ``field_name``
256 will be traversed across relationships. Returns '[invalid name]' for
257 any field name that cannot be traversed.
258  
259 ex::
260  
261 >>> verbose_field_name(Article, 'author__name')
262 'author name'
263  
264 """
265 if field_name is None:
266 return "[invalid name]"

269 if not parts:
270 return "[invalid name]"

275 if part.related_name:
276 names.append(part.related_name.replace("_", " "))
277 else:
278 return "[invalid name]"
279 else:
280 names.append(force_str(part.verbose_name))
281
