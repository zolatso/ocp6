97 """
98 Determine if two paths reference the same file.
99  
100 Augments os.path.samefile to work on Windows and
101 suppresses errors if the path doesn't exist.
102 """
103 both_exist = os.path.exists(p1) and os.path.exists(p2)
104 use_samefile = hasattr(os.path, "samefile") and both_exist
105 if use_samefile:
106 return os.path.samefile(p1, p2)
107 norm_p1 = os.path.normpath(os.path.normcase(p1))
108 norm_p2 = os.path.normpath(os.path.normcase(p2))
109 return norm_p1 == norm_p2
