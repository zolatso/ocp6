21 def direct_url_as_pep440_direct_reference(direct_url, name):
22 # type: (DirectUrl, str) -> str
23 """Convert a DirectUrl to a pip requirement string."""
24 direct_url.validate() # if invalid, this is a pip bug
25 requirement = name + " @ "
26 fragments = []
27 if isinstance(direct_url.info, VcsInfo):
28 requirement += "{}+{}@{}".format(
29 direct_url.info.vcs, direct_url.url, direct_url.info.commit_id
30 )
31 elif isinstance(direct_url.info, ArchiveInfo):
32 requirement += direct_url.url
33 if direct_url.info.hash:
34 fragments.append(direct_url.info.hash)
35 else:
36 assert isinstance(direct_url.info, DirInfo)
37 requirement += direct_url.url
38 if direct_url.subdirectory:
39 fragments.append("subdirectory=" + direct_url.subdirectory)
40 if fragments:
41 requirement += "#" + "&".join(fragments)
42 return requirement

45 def direct_url_from_link(link, source_dir=None, link_is_in_wheel_cache=False):
46 # type: (Link, Optional[str], bool) -> DirectUrl
47 if link.is_vcs:
48 vcs_backend = vcs.get_backend_for_scheme(link.scheme)
49 assert vcs_backend
50 url, requested_revision, _ = vcs_backend.get_url_rev_and_auth(
51 link.url_without_fragment
52 )
53 # For VCS links, we need to find out and add commit_id.
54 if link_is_in_wheel_cache:
55 # If the requested VCS link corresponds to a cached
56 # wheel, it means the requested revision was an
57 # immutable commit hash, otherwise it would not have
58 # been cached. In that case we don't have a source_dir
59 # with the VCS checkout.
60 assert requested_revision
61 commit_id = requested_revision
62 else:
63 # If the wheel was not in cache, it means we have
64 # had to checkout from VCS to build and we have a source_dir
65 # which we can inspect to find out the commit id.
66 assert source_dir
67 commit_id = vcs_backend.get_revision(source_dir)
68 return DirectUrl(
69 url=url,
70 info=VcsInfo(
71 vcs=vcs_backend.name,
72 commit_id=commit_id,
73 requested_revision=requested_revision,
74 ),
75 subdirectory=link.subdirectory_fragment,
76 )
77 elif link.is_existing_dir():
78 return DirectUrl(
79 url=link.url_without_fragment,
80 info=DirInfo(),
81 subdirectory=link.subdirectory_fragment,
82 )
83 else:
84 hash = None
85 hash_name = link.hash_name
86 if hash_name:
87 hash = f"{hash_name}={link.hash}"
88 return DirectUrl(
89 url=link.url_without_fragment,
90 info=ArchiveInfo(hash=hash),
91 subdirectory=link.subdirectory_fragment,
92 )

97 """Obtain a DirectUrl from a pkg_resource.Distribution.
98  
99 Returns None if the distribution has no `direct_url.json` metadata,
100 or if `direct_url.json` is invalid.
101 """
102 if not dist.has_metadata(DIRECT_URL_METADATA_NAME):
103 return None
104 try:
105 return DirectUrl.from_json(dist.get_metadata(DIRECT_URL_METADATA_NAME))
106 except (
107 DirectUrlValidationError,
108 json.JSONDecodeError,
109 UnicodeDecodeError,
110 ) as e:
111 logger.warning(
112 "Error parsing %s for %s: %s",
113 DIRECT_URL_METADATA_NAME,
114 dist.project_name,
115 e,
116 )
117 return None
