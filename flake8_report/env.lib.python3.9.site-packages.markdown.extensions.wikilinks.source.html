30 def build_url(label, base, end):
31 """ Build a URL from the label, a base, and an end. """
32 clean_label = re.sub(r'([ ]+_)|(_[ ]+)|([ ]+)', '_', label)
33 return '{}{}{}'.format(base, clean_label, end)

66 def handleMatch(self, m, data):
67 if m.group(1).strip():
68 base_url, end_url, html_class = self._getMeta()
69 label = m.group(1).strip()
70 url = self.config['build_url'](label, base_url, end_url)
71 a = etree.Element('a')
72 a.text = label
73 a.set('href', url)
74 if html_class:
75 a.set('class', html_class)
76 else:
77 a = ''
78 return a, m.start(0), m.end(0)
