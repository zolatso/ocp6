242 def _get_user_dirs_folder(key: str) -> str | None:
243 """
244 Return directory from user-dirs.dirs config file.
245  
246 See https://freedesktop.org/wiki/Software/xdg-user-dirs/.
247  
248 """
249 user_dirs_config_path = Path(Unix().user_config_dir) / "user-dirs.dirs"
250 if user_dirs_config_path.exists():
251 parser = ConfigParser()
252  
253 with user_dirs_config_path.open() as stream:
254 # Add fake section header, so ConfigParser doesn't complain
255 parser.read_string(f"[top]\n{stream.read()}")
256  
257 if key not in parser["top"]:
258 return None
259  
260 path = parser["top"][key].strip('"')
261 # Handle relative home paths
262 return path.replace("$HOME", os.path.expanduser("~")) # noqa: PTH111
