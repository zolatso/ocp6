83 """ Parse a table block and build table. """
84 block = blocks.pop(0).split('\n')
85 header = block[0].strip(' ')
86 rows = [] if len(block) < 3 else block[2:]

88 # Get alignment of columns
89 align = []
90 for c in self.separator:
91 c = c.strip(' ')
92 if c.startswith(':') and c.endswith(':'):
93 align.append('center')
94 elif c.startswith(':'):
95 align.append('left')
96 elif c.endswith(':'):
97 align.append('right')
98 else:

102 table = etree.SubElement(parent, 'table')
103 thead = etree.SubElement(table, 'thead')
104 self._build_row(header, thead, align)
105 tbody = etree.SubElement(table, 'tbody')
106 if len(rows) == 0:

121 def _build_row(self, row, parent, align):
122 """ Given a row of text, build table cells. """
123 tr = etree.SubElement(parent, 'tr')
124 tag = 'td'
125 if parent.tag == 'thead':
126 tag = 'th'
127 cells = self._split_row(row)
128 # We use align here rather than cells to ensure every row
129 # contains the same number of columns.
130 for i, a in enumerate(align):
131 c = etree.SubElement(tr, tag)
132 try:
133 c.text = cells[i].strip(' ')
134 except IndexError: # pragma: no cover
135 c.text = ""
136 if a:
137 if self.config['use_align_attribute']:
138 c.set('align', a)
139 else:
