106 @staticmethod
107 def _copy_permissions(mode, filename):
108 """
109 If the file in the archive has some permissions (this assumes a file
110 won't be writable/executable without being readable), apply those
111 permissions to the unarchived file.
112 """
113 if mode & stat.S_IROTH:
114 os.chmod(filename, mode)

116 def split_leading_dir(self, path):
117 path = str(path)
118 path = path.lstrip("/").lstrip("\\")
119 if "/" in path and (
  • E501 Line too long (83 > 79 characters)
120 ("\\" in path and path.find("/") < path.find("\\")) or "\\" not in path
121 ):
122 return path.split("/", 1)
123 elif "\\" in path:
124 return path.split("\\", 1)
125 else:
126 return path, ""

128 def has_leading_dir(self, paths):
129 """
130 Return True if all the paths have the same leading path name
131 (i.e., everything is in one subdirectory in an archive).
132 """
133 common_prefix = None
134 for path in paths:
135 prefix, rest = self.split_leading_dir(path)
136 if not prefix:
137 return False
138 elif common_prefix is None:
139 common_prefix = prefix
140 elif prefix != common_prefix:
141 return False
142 return True

169 def extract(self, to_path):
170 members = self._archive.getmembers()
171 leading = self.has_leading_dir(x.name for x in members)
172 for member in members:
173 name = member.name
174 if leading:
175 name = self.split_leading_dir(name)[1]
176 filename = self.target_filename(to_path, name)
177 if member.isdir():
178 if filename:
179 os.makedirs(filename, exist_ok=True)
180 else:
181 try:
182 extracted = self._archive.extractfile(member)
183 except (KeyError, AttributeError) as exc:
184 # Some corrupt tar files seem to produce this
185 # (specifically bad symlinks)
186 print(
187 "In the tar file %s the member %s is invalid: %s"
188 % (name, member.name, exc)
189 )
190 else:
191 dirname = os.path.dirname(filename)
192 if dirname:
193 os.makedirs(dirname, exist_ok=True)
194 with open(filename, "wb") as outfile:
195 shutil.copyfileobj(extracted, outfile)
196 self._copy_permissions(member.mode, filename)
197 finally:
198 if extracted:
199 extracted.close()
