15 def glibc_version_string_confstr():
16 # type: () -> Optional[str]
17 "Primary implementation of glibc_version_string using os.confstr."
18 # os.confstr is quite a bit faster than ctypes.DLL. It's also less likely
19 # to be broken or missing. This strategy is used in the standard library
20 # platform module:
21 # https://github.com/python/cpython/blob/fcf1d003bf4f0100c9d0921ff3d70e1127ca1b71/Lib/platform.py#L175-L183
22 if sys.platform == "win32":
23 return None
24 try:
25 # os.confstr("CS_GNU_LIBC_VERSION") returns a string like "glibc 2.17":
26 _, version = os.confstr("CS_GNU_LIBC_VERSION").split()
27 except (AttributeError, OSError, ValueError):
28 # os.confstr() or CS_GNU_LIBC_VERSION not available (or a bad value)...
29 return None
30 return version

33 def glibc_version_string_ctypes():
34 # type: () -> Optional[str]
35 "Fallback implementation of glibc_version_string using ctypes."
36  
37 try:
38 import ctypes
39 except ImportError:
40 return None

42 # ctypes.CDLL(None) internally calls dlopen(NULL), and as the dlopen
43 # manpage says, "If filename is NULL, then the returned handle is for the
44 # main program". This way we can let the linker do the work to figure out
45 # which libc our process is actually using.
46 process_namespace = ctypes.CDLL(None)
47 try:
48 gnu_get_libc_version = process_namespace.gnu_get_libc_version
49 except AttributeError:
50 # Symbol doesn't exist -> therefore, we are not linked to
51 # glibc.
52 return None

54 # Call gnu_get_libc_version, which returns a string like "2.5"
55 gnu_get_libc_version.restype = ctypes.c_char_p
56 version_str = gnu_get_libc_version()
57 # py2 / py3 compatibility:
58 if not isinstance(version_str, str):
59 version_str = version_str.decode("ascii")

61 return version_str

67 # ~$ python2.7 -c 'import platform; print(platform.libc_ver())'

78 # versions that was generated by pip 8.1.2 and earlier is useless and
79 # misleading. Solution: instead of using platform, use our code that actually
80 # works.
