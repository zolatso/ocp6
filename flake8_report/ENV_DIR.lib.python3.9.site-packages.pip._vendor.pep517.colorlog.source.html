29 def _stderr_supports_color():
30 color = False
31 if curses and hasattr(sys.stderr, "isatty") and sys.stderr.isatty():
32 try:
33 curses.setupterm()
34 if curses.tigetnum("colors") > 0:
35 color = True
36 except Exception:
37 pass
38 return color

41 class LogFormatter(logging.Formatter):
42 """Log formatter with colour support"""
43  
44 DEFAULT_COLORS = {
45 logging.INFO: 2, # Green
46 logging.WARNING: 3, # Yellow
47 logging.ERROR: 1, # Red
48 logging.CRITICAL: 1,
49 }
50  
51 def __init__(self, color=True, datefmt=None):
52 r"""
53 :arg bool color: Enables color support.
54 :arg string fmt: Log message format.
55 It will be applied to the attributes dict of log records. The
56 text between ``%(color)s`` and ``%(end_color)s`` will be colored
57 depending on the level if color support is on.
58 :arg dict colors: color mappings from logging level to terminal color
59 code
60 :arg string datefmt: Datetime format.
61 Used for formatting ``(asctime)`` placeholder in ``prefix_fmt``.
62 .. versionchanged:: 3.2
63 Added ``fmt`` and ``datefmt`` arguments.
64 """
65 logging.Formatter.__init__(self, datefmt=datefmt)
66 self._colors = {}
67 if color and _stderr_supports_color():
68 # The curses module has some str/bytes confusion in
69 # python3. Until version 3.2.3, most methods return
70 # bytes, but only accept strings. In addition, we want to
71 # output these strings with the logging module, which
72 # works with unicode strings. The explicit calls to
73 # unicode() below are harmless in python2 but will do the
74 # right conversion in python 3.
  • E501 Line too long (80 > 79 characters)
75 fg_color = curses.tigetstr("setaf") or curses.tigetstr("setf") or ""
76 if (3, 0) < sys.version_info < (3, 2, 3):
77 fg_color = str(fg_color, "ascii")
78  
79 for levelno, code in self.DEFAULT_COLORS.items():
  • E501 Line too long (82 > 79 characters)
80 self._colors[levelno] = str(curses.tparm(fg_color, code), "ascii")
81 self._normal = str(curses.tigetstr("sgr0"), "ascii")
82  
83 scr = curses.initscr()
84 self.termwidth = scr.getmaxyx()[1]
85 curses.endwin()
86 else:
87 self._normal = ""
88 # Default width is usually 80, but too wide is
89 # worse than too narrow
90 self.termwidth = 70

92 def formatMessage(self, record):
93 mlen = len(record.message)
94 right_text = "{initial}-{name}".format(
95 initial=record.levelname[0], name=record.name
96 )
97 if mlen + len(right_text) < self.termwidth:
98 space = " " * (self.termwidth - (mlen + len(right_text)))
99 else:
100 space = " "
101  
102 if record.levelno in self._colors:
103 start_color = self._colors[record.levelno]
104 end_color = self._normal
105 else:
106 start_color = end_color = ""
107  
108 return record.message + space + start_color + right_text + end_color

111 def enable_colourful_output(level=logging.INFO):
112 handler = logging.StreamHandler()
113 handler.setFormatter(LogFormatter())
114 logging.root.addHandler(handler)
115 logging.root.setLevel(level)
