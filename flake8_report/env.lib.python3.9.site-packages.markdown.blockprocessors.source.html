607 def run(self, parent, blocks):
608 block = blocks.pop(0)
609 if block.strip():
610 # Not a blank block. Add to parent, otherwise throw it away.
611 if self.parser.state.isstate('list'):
612 # The parent is a tight-list.
613 #
614 # Check for any children. This will likely only happen in a
615 # tight-list when a header isn't followed by a blank line.
616 # For example:
617 #
618 # * # Header
619 # Line 2 of list item - not part of header.
620 sibling = self.lastChild(parent)
621 if sibling is not None:
622 # Insert after sibling.
623 if sibling.tail:
624 sibling.tail = '{}\n{}'.format(sibling.tail, block)
625 else:
626 sibling.tail = '\n%s' % block
627 else:
628 # Append to parent.text
629 if parent.text:
630 parent.text = '{}\n{}'.format(parent.text, block)
631 else:
632 parent.text = block.lstrip()
633 else:
634 # Create a regular paragraph
635 p = etree.SubElement(parent, 'p')
636 p.text = block.lstrip()
