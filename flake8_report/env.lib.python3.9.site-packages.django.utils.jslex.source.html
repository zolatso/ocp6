50 def lex(self, text):
51 """
52 Lexically analyze `text`.
53  
54 Yield pairs (`name`, `tokentext`).
55 """
56 end = len(text)
57 state = self.state
58 regexes = self.regexes
59 toks = self.toks
60 start = 0
61  
62 while start < end:
63 for match in regexes[state].finditer(text, start):
64 name = match.lastgroup
65 tok = toks[name]
66 toktext = match[name]
67 start += len(toktext)
68 yield (tok.name, toktext)
69  
70 if tok.next:
71 state = tok.next
72 break
73  
74 self.state = state

213 def prepare_js_for_gettext(js):
214 """
215 Convert the JavaScript source `js` into something resembling C for
216 xgettext.
217  
218 What actually happens is that all the regex literals are replaced with
219 "REGEX".
220 """
221  
222 def escape_quotes(m):
223 """Used in a regex to properly escape double quotes."""
224 s = m[0]
225 if s == '"':
226 return r"\""
227 else:
228 return s
229  
230 lexer = JsLexer()
231 c = []
232 for name, tok in lexer.lex(js):
233 if name == "regex":
234 # C doesn't grok regexes, and they aren't needed for gettext,
235 # so just output a string instead.
236 tok = '"REGEX"'
237 elif name == "string":
238 # C doesn't have single-quoted strings, so make all strings
239 # double-quoted.
240 if tok.startswith("'"):
241 guts = re.sub(r"\\.|.", escape_quotes, tok[1:-1])
242 tok = '"' + guts + '"'
243 elif name == "id":
244 # C can't deal with Unicode escapes in identifiers. We don't
245 # need them for gettext anyway, so replace them with something
246 # innocuous
247 tok = tok.replace("\\", "U")
248 c.append(tok)
249 return "".join(c)
